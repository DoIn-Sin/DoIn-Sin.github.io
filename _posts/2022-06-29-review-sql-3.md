# SELECT 더 깊이 파보기 - 서브쿼리

## 1. 비상관 서브쿼리
- 사용할 서브쿼리가 외부 쿼리의 어떤 것도 참조하지 않고 단독으로 사용되는 경우

```sql
-- Categories 테이블에서 CategoryID, CategoryName, Description을 가져오고,
-- Products 테이블에서 ProductID가 1인 ProductName도 같이 가져와서 출력
SELECT
  CategoryID, CategoryName, Description,
  (SELECT ProductName FROM Products WHERE ProductID = 1)
FROM Categories;

-- Producs 모든 컬럼을 출력하는데, 테이블의 Price가 평균보다 작은 row들만 가져와 출력
SELECT * FROM Products
WHERE Price < (
  SELECT AVG(Price) FROM Products
);

-- Categoris 테이블에서 CategoryID, CategoryName, Description을 가져오는데,
-- CategortID가 Products 테이블의 ProductName이 'Chais'인 CategoryID와 같은 것만 가져와 출력
SELECT
  CategoryID, CategoryName, Description
FROM Categories
WHERE
  CategoryID =
  (SELECT CategoryID FROM Products
  WHERE ProductName = 'Chais');

-- Categoris 테이블에서 CategoryID, CategoryName, Description을 가져오는데,
-- CategoryID가 Products 테이블의 Price가 50보다 큰 것들의 CategoryID들에 속하는 것만 가져와 출력
SELECT
  CategoryID, CategoryName, Description
FROM Categories
WHERE
  CategoryID IN
  (SELECT CategoryID FROM Products
  WHERE Price > 50);
```

-- **ALL(Sub Query)**: 서브쿼리의 모든 결과에 대해 검토, **ANY(Sub Query)**: 서브쿼리의 하나 이상의 결과에 대해 검토

```sql
-- Products 테이블의 모든 컬럼을 가져오는데, CategoryID가 2인 것의 Price보다 큰 Price의 것들만 가져와 출력
SELECT * FROM Products
WHERE Price > ALL (
  SELECT Price FROM Products
  WHERE CategoryID = 2
);

-- Categoris 테이블에서 CategoryID, CategoryName, Description을 가져오는데,
-- CategoryID가 Products 테이블의 Price가 50보다 큰 것들의 CategoryID들 중에 똑같은 것들만 가져와 출력
SELECT
  CategoryID, CategoryName, Description
FROM Categories
WHERE
  CategoryID = ANY
  (SELECT CategoryID FROM Products
  WHERE Price > 50);
```

## 2. 상관 서브쿼리
- 내부 쿼리의 값이 결정 되는데 외부 쿼리에 의존하는 경우

```sql
-- Products 테이블의 ProductID, ProductName을 가져오고, 
-- CategoryID는 Categories 테이블의 CategoryID과 같은 것만 가져와 출력
SELECT
  ProductID, ProductName,
  (
    SELECT CategoryName FROM Categories C
    WHERE C.CategoryID = P.CategoryID
  ) AS CategoryName
FROM Products P;

SELECT
  SupplierName, Country, City,
  (
    SELECT COUNT(*) FROM Customers C
    WHERE C.Country = S.Country
  ) AS CustomersInTheCountry,
  (
    SELECT COUNT(*) FROM Customers C
    WHERE C.Country = S.Country 
      AND C.City = S.City
  ) AS CustomersInTheCity
FROM Suppliers S;

SELECT
  CategoryID, CategoryName,
  (
    SELECT MAX(Price) FROM Products P
    WHERE P.CategoryID = C.CategoryID
  ) AS MaximumPrice,
  (
    SELECT AVG(Price) FROM Products P
    WHERE P.CategoryID = C.CategoryID
  ) AS AveragePrice
FROM Categories C;

SELECT
  ProductID, ProductName, CategoryID, Price
  -- ,(SELECT AVG(Price) FROM Products P2
  -- WHERE P2.CategoryID = P1.CategoryID)
FROM Products P1
WHERE Price < (
  SELECT AVG(Price) FROM Products P2
  WHERE P2.CategoryID = P1.CategoryID
);
```

- **EXISTS(Sub Query)**: 해당되는 내용의 row만 가져옴
- NOT과 같이 쓸 수 있다.

```sql
SELECT
  CategoryID, CategoryName
  -- ,(SELECT MAX(P.Price) FROM Products P
  -- WHERE P.CategoryID = C.CategoryID
  -- ) AS MaxPrice
FROM Categories C
WHERE EXISTS (
  SELECT * FROM Products P
  WHERE P.CategoryID = C.CategoryID
  AND P.Price > 80
);
```
